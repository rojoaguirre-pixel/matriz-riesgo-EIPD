    <script>

        function simularGeneracion() {

            const input = document.getElementById('userInput').value;

            if(!input) return alert("Por favor, ingresa un contexto.");


            document.getElementById('emptyState').classList.add('hidden');

            document.getElementById('resultContainer').classList.add('hidden');

            document.getElementById('loader').classList.remove('hidden');


            // Simulación de respuesta de Gemini Pro 3.1

            setTimeout(() => {

                const riesgos = [

                    { nombre: "Acceso no autorizado a datos biométricos", prob: "Baja", imp: "Muy Alto", nivel: "Crítico" },

                    { nombre: "Tratamiento sin base legal adecuada", prob: "Media", imp: "Alto", nivel: "Alto" },

                    { nombre: "Falta de transparencia en el aviso de privacidad", prob: "Alta", imp: "Medio", nivel: "Medio" }

                ];


                const tbody = document.getElementById('matrizBody');

                tbody.innerHTML = "";


                riesgos.forEach(r => {

                    const color = r.nivel === 'Crítico' ? 'bg-red-100 text-red-700' : (r.nivel === 'Alto' ? 'bg-orange-100 text-orange-700' : 'bg-yellow-100 text-yellow-700');

                    tbody.innerHTML += `

                        <tr class="border-b border-slate-100 hover:bg-slate-50 transition-colors">

                            <td class="p-4 text-sm font-medium text-slate-700">${r.nombre}</td>

                            <td class="p-4 text-sm text-center text-slate-600">${r.prob}</td>

                            <td class="p-4 text-sm text-center text-slate-600">${r.imp}</td>

                            <td class="p-4 text-center">

                                <span class="px-3 py-1 rounded-full text-xs font-bold ${color}">${r.nivel}</span>

                            </td>

                        </tr>

                    `;

                });


                document.getElementById('loader').classList.add('hidden');

                document.getElementById('resultContainer').classList.remove('hidden');

            }, 2000);

        }
